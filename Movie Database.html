desc movies; /* genres table creation*/ create table genre(genre_id varchar(10) primary key, genre_name varchar(20)); desc genre; /* movie genres table creation*/ create table movie_genres(movie_id varchar(10),genre_id varchar(10),foreign key(movie_id) references movies(movie_id),foreign key(genre_id) references genre(genre_id));
